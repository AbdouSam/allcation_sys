333 static int sram_alloc_op(int alloc_id, uint32_t pos, void *buffer, uint32_t len, bool write)
334 {
335  int i;
336  int size;
337 
338  superblock_t *superblock = &gsuper_block;
339  block_t *blocktable = gblock_table;
340  alloc_t *alloctable = galloc_table;
341 
342  uint32_t blockoffset;
343  uint32_t posoffset;
344  uint32_t nbrblocks;
345  uint32_t blocktrim;
346  uint32_t trim;
347  uint32_t bufflen;
348  uint32_t lastlen;
349  uint8_t *localbuffer = local_buffer;
350  uint32_t nextblock;
351 
352  /* This also checks if id is allocated. */
353  size = sram_getalloc_size(alloc_id);
354 
355  if(size < 0)
356  {
357  return size;
358  }
359 
360  /* check whether position is in the range of writing. */
361  if ((len + pos) > size)
362  {
363  /* length of data exceed allocated space */
364  return ERR_ALLOC_RW_EXCEED;
365  }
366 
367  /* Determine how many blocks are to be written */
368  blockoffset = (pos / BLOCK_SIZE);
369  posoffset = (pos % BLOCK_SIZE);
370  nbrblocks = ((posoffset + len) / BLOCK_SIZE) + (((posoffset + len) % BLOCK_SIZE) > 0);
371 
372  if ((len + pos) > BLOCK_SIZE)
373  {
374  blocktrim = (BLOCK_SIZE - ((len + pos) % BLOCK_SIZE)) % BLOCK_SIZE;
375  }
376  else
377  {
378  blocktrim = BLOCK_SIZE - len - pos;
379  }
380 
381  /* first block to operate. */
382  nextblock = alloctable[alloc_id].start;
383 
384  /* Offset from the first block. */
385  for (i = blockoffset; i > 0; i--)
386  {
387  nextblock = blocktable[nextblock].next;
388  }
389 
390  dbg_msg("%s :blockoffset %d, firstposoffset %d, nbrblockstowrite %d, trim %d\n",
391  ((write)?"Write" : "Read"), blockoffset, posoffset, nbrblocks, blocktrim);
392 
393  trim = 0;
394  lastlen = 0;
395 
396  for (i = nbrblocks; i > 0; i--)
397  {
398  /* Read the buffer locally. */
399  sram_onblock_op(localbuffer, nextblock, 1, ALLOC_OP_READ);
400 
401  if (i == 1)
402  {
403  trim = blocktrim;
404  }
405 
406  bufflen = BLOCK_SIZE - posoffset - trim;
407 
408  dbg_msg("blocks number = %d, len %d, off %d\n", nextblock, bufflen, posoffset);
409 
410  if (write)
411  {
412  /* OverWrite locally by buffer data. */
413  memcpy(localbuffer + posoffset, buffer + lastlen, bufflen);
414 
415  /* Write the block back */
416  sram_onblock_op(localbuffer, nextblock, 1, ALLOC_OP_WRITE);
417  }
418  else
419  {
420  /* copy to read buffer.*/
421  memcpy(buffer + lastlen, localbuffer + posoffset, bufflen);
422  }
423 
424  /* Copy to the buffer from last length. */
425  lastlen += bufflen;
426 
427  /* Jump to next block */
428  nextblock = blocktable[nextblock].next;
429 
430  /* offset serves only at the begining. */
431  posoffset = 0;
432  }
433 
434  return ALLOC_OK;
435 }

462  superblock->datablocknbr = DATA_BLKNBR;/* the reste of blocks */
463  superblock->spstrindex = SUPER_STR_INDX;
464  superblock->btstrindex = BLOCK_TABLE_STR_INDX;
465  superblock->alstrindex = ALLOC_TABLE_STR_INDX;
466  superblock->datastrindex = DATA_STR_INDX;

468  superblock->alentrynbr = ALLOC_TABLE_ROW_NBR;
469  superblock->allocnbr = 0; /* No allocation yet. */

473  superblock->magicnbr, superblock->blocknbr, superblock->alentrynbr);
474

477  init_tables();
478  }

489 {
490  int i, j;

492  superblock_t *superblock = &gsuper_block;
493  block_t *blocktable = gblock_table;
494  alloc_t *alloctable = galloc_table;
495 
496  uint8_t blocknbr;
497  int firstblock = 0;
498  int blockcount;
499 
500  if (bytenbr < 0)
501  {
502  /* bytenbr invalid */
503  return ERR_ALLOC_INVALID_BYTENBR;
504  }
505  else if (bytenbr <= BLOCK_SIZE)
506  {
507  blocknbr = 1;
508  }
509  else
510  {
511  blocknbr = (bytenbr / BLOCK_SIZE) + ((bytenbr%BLOCK_SIZE) > 0);
512  }
513 
514  /* Check if system is mounted properly. */
515  if (superblock->magicnbr != (uint32_t)ALLOC_MAGIC_NBR)
516  {
517  return ERR_ALLOC_SYS_NOTFOUND;
518  }
519 
520  if (superblock->allocnbr > ALLOC_TABLE_ROW_NBR)
521  {
522  /* Allocation table full. */
523  return ERR_ALLOC_TABLE_FULL;
524  }
525 
526  if (superblock->freeblocknbr < blocknbr)
527  {
528  /* Allocation failed not enough blocks. */
529  return ERR_BLOCK_TABLE_FULL;
530  }
531 
532  if (alloctable[alloc_id].used == 1)
533  {
534  return ERR_ALLOC_USED;
535  }
536 
537  if ((alloc_id > superblock->alentrynbr) || (alloc_id < 0))
538  {
539  return ERR_ALLOC_ID_INVALID;
540  }
541 
542  /* Search for free blocks */
543  blockcount = blocknbr;
544  for (i = superblock->datastrindex; i < superblock->blentrynbr; i++)
545  {
546  if (blocktable[i].busy != (uint8_t)1)
547  {
548  /* Allocation found. */
549  if (!firstblock)
550  {
551  firstblock = i;
552  }
553  else
554  {
555  blocktable[lastindex].next = i;
556  }
557 
558  blocktable[i].busy = 1;
559 
560  blockcount--;
561  if (blockcount == 0)
562  {
563  blocktable[i].next = -1;
564 
565  alloctable[alloc_id].used = 1;
566  alloctable[alloc_id].start = firstblock;
567  alloctable[alloc_id].len = blocknbr;
568 
569  update_alloc_superblock(superblock, blocknbr);
570 
571  sync_iblocks_sram(superblock, blocktable, alloctable);
572 
573  break;
574  }
575  lastindex = i;
576  }
577  }
578  if (i > superblock->blentrynbr)
579  {
580  return ERR_ALLOC_BLOCKS_FULL;
581  }
582 
583  if (blockcount != 0)
584  {
585  /* Allocation interrupted.*/
586  return ERR_ALLOC_FAILED;
587  }
588 
589  return ALLOC_OK;
590 }
